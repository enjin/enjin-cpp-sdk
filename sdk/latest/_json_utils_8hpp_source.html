83  template<class T>
84  static bool try_get_array_as_type_array(const json::JsonValue& json, const std::string& key, std::vector<T>& out) {
85  static_assert(std::is_base_of_v<serialization::IDeserializable, T>,
86  "Class T does not inherit from IDeserializable.");
87 
88  if (!json.is_object()) {
89  return false;
90  }
91 
92  json::JsonValue array;
93  if (!json.try_get_object_field(key, array)) {
94  return false;
95  }
96 
97  std::vector<json::JsonValue> value_array;
98  if (!array.try_get_array(value_array)) {
99  return false;
100  }
101 
102  out.clear();
103 
104  for (const json::JsonValue& value: value_array) {
105  T t;
106  t.deserialize(value.to_string());
107 
108  out.push_back(std::move(t));
109  }
110 
111  return true;
112  }

163  template<class T>
164  static bool try_get_object_as_type(const json::JsonValue& json, const std::string& key, T& out) {
165  static_assert(std::is_base_of_v<serialization::IDeserializable, T>,
166  "Class T does not inherit from IDeserializable.");
167 
168  json::JsonValue value = json::JsonValue::create_object();
169  if (!json.try_get_object_field(key, value) || !value.is_object()) {
170  return false;
171  }
172 
173  out.deserialize(value.to_string());
174  return true;
175  }

225 template<>
226 inline json::JsonValue JsonUtils::get_type_array_as_json_value_array(const std::vector<std::string>& array) {
227  json::JsonValue value_array = json::JsonValue::create_array();
228 
229  for (const std::string& el: array) {
230  json::JsonValue value_string = json::JsonValue::create_string();
231 
232  if (value_string.try_set_string(el)) {
233  value_array.try_set_array_element(value_string);
234  }
235  }
236 
237  return value_array;
